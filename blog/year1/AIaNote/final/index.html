def bfs(a_cap, b_cap, target):
    visited = set()
    queue = deque()
    queue.append((0, 0))  # ÂàùÂßãÁä∂ÊÄÅÔºå‰∏§‰∏™Â£∂ÈÉΩÁ©∫
    visited.add((0, 0))

    while queue:
        x, y = queue.popleft()
        if x == target or y == target:
            return True  # ÊâæÂà∞Ëß£

        # ÊâÄÊúâÂèØËÉΩÁöÑÂÄíÊ∞¥Êìç‰Ωú
        next_states = [

            (x, b_cap),  # B ÂÄíÊª°
            (0, y),      # A ÂÄíÁ©∫
            (x, 0),      # B ÂÄíÁ©∫
            # A -> B
            (max(0, x - (b_cap - y)), min(b_cap, x + y)),
            # B -> A
            (min(a_cap, x + y), max(0, y - (a_cap - x)))
        ]

        for nx, ny in next_states:
            if (nx, ny) not in visited:
                visited.add((nx, ny))
                queue.append((nx, ny))

    return False
