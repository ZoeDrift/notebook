    def fib(n):
    if n == 0 or n == 1:
        return 1
    a, b = 1, 1
    for i in range(2, n + 1):
        a, b = b, a + b
    return b

def PrintFN(m, n):
    fibs = []
    a, b = 1, 1
    while a <= n:
        if a >= m:
            fibs.append(a)
        a, b = b, a + b
    return fibs
